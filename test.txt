function loadShopInfo(slug) {
  currentShopInfo = null;
  setShopHeader(null);
  if (!slug) return;

  fetch(`http://127.0.0.1:8000/shops/${encodeURIComponent(slug)}`)
    .then(r => r.ok ? r.json() : Promise.reject())
    .then(info => {
      currentShopInfo = info;
      setShopHeader(info);
    })
    .catch(() => {
      currentShopInfo = null;
      setShopHeader(null);
    });
}
